166  virtual ~btosgObject() {
167  if (body) {
168  delete body->getMotionState();
169  delete body;
170  body = NULL;
171  }
172  if( model ) {
173  model->unref();
174  model = NULL;
175  }
176  if (shape) {
177  delete shape;

187  name = strdup(n);

189  void setMass(double m) {
191  mass = m;
192  bool isDynamic = (mass != 0.f);
193 
194  btVector3 localInertia(0, 0, 0);
195  if (isDynamic && shape)
196  shape->calculateLocalInertia(mass, localInertia);
197  if ( body ) body->setMassProps(m, localInertia);
198  }
199  btVector3 getPosition() {
201  if (body) {
202  btTransform wTrans;
203  body->getMotionState()->getWorldTransform(wTrans);
204  return wTrans.getOrigin();
205  }
206  if (model) return osg2bt_Vec3(model->getPosition());
207  return btVector3(0.,0.,0.);
208  }
209  btQuaternion getRotation() {
211  if (body) {
212  btTransform wTrans;
213  body->getMotionState()->getWorldTransform(wTrans);
214  return wTrans.getRotation();
215  }
216  if (model) return osg2bt_Quat(model->getAttitude());
217  return btQuaternion(0.,0.,0.,0.);
218  }
219  btVector3 getEuler() {
221  btQuaternion qt = getRotation();
222  return quat2Euler(qt);
223  }
224  void setPosition(const btVector3 &p) {
226  if (body) {
227  btTransform wTrans;
228  body->getMotionState()->getWorldTransform(wTrans);
229  wTrans.setOrigin(p);
230  body->setWorldTransform(wTrans);
231  body->getMotionState()->setWorldTransform(wTrans);
232  body->clearForces();
233  body->setLinearVelocity(btVector3(0,0,0));
234  body->setAngularVelocity(btVector3(0,0,0));
235  }
236  #ifdef AVOID_DIRECT_MODEL_UPDATE
237  else
238  #endif
239  { // Not required for dynamic objects.
240  #ifdef _DEBUG_
241  printf("set Position in non dynamic object\n");
242  #endif
243  if (model) {
244  model->setPosition(bt2osg_Vec3(p));
245  }
246  }
247  }

255  btTransform wTrans;

262  body->setLinearVelocity(btVector3(0,0,0));

300  virtual void update() {
304  if (body) {
305  btTransform wTrans;
306  body->getMotionState()->getWorldTransform(wTrans);
307  if ( model ) {
308  model->setAttitude(bt2osg_Quat(wTrans.getRotation()));
309  model->setPosition(bt2osg_Vec3(wTrans.getOrigin()));
310  }
311  }
312  }
313  void reset() {
315  if ( body ) {
316  body->setWorldTransform(init_state);
317  body->getMotionState()->setWorldTransform(init_state);
318  body->clearForces();
319  if ( mass>0. ) {
320  body->setLinearVelocity(btVector3(0.,0.,0.));
321  body->setAngularVelocity(btVector3(0.,0.,0.));
322  body->activate(); // Required if the object was asleep
323  }
324  }
325  else { // Not required for dynamic objects.
326  if ( model ) {
327  model->setAttitude(bt2osg_Quat(init_state.getRotation()));
328  model->setPosition(bt2osg_Vec3(init_state.getOrigin()));
329  }
330  }
331  }
332  void setInitState() {
335  if (body) body->getMotionState()->getWorldTransform(init_state);
336  }
337  void setInitState(btTransform iState) {
340  init_state = iState;
341  }
342  void createRigidBody() {
344  if ( ! shape ) {
345  fprintf(stderr,"Cannot create RigidBody without shape\n");
346  return;
347  }
348  btDefaultMotionState* mState = new
349  btDefaultMotionState(btTransform(btQuaternion(0,0,0,1),btVector3(0.,0.,0.)));
350  btVector3 inertia(0,0,0);
351  shape->calculateLocalInertia(mass,inertia);
352  btRigidBody::btRigidBodyConstructionInfo cInfo(mass,mState,shape,inertia);
353  //printf("mass: %f\n",mass);
354  cInfo.m_restitution = 0.9f;
355  cInfo.m_friction = 10.f;
356  body = new btRigidBody(cInfo);
357  if ( !body ) fprintf(stderr,"Error creating btBody\n");
358  }
