57  btosgVehicle(btosgWorld *world, osg::Vec3 dimLocal = osg::Vec3(2.,0.4,4.), double m=1000. ) {
58  btVector3 grav = world->dynamic->getGravity();
59  int grav_axis = grav.minAxis();
60  switch ( grav_axis ) {
61  case 0:
62  fprintf(stderr,"Gravity direction %d (%f,%f,%f) not supported\n",grav_axis,grav[0],grav[1],grav[2]);
63  break;
64  case 1:
65  up = osg::Vec3(0., 1., 0.);
66  front = osg::Vec3(0., 0., 1.);
67  break;
68  case 2:
69  up = osg::Vec3(0., 0., 1.);
70  front = osg::Vec3(0., 1., 0.);
71  break;
72  }
73  // dimLocal contains (Width,Height,Lenght)
74  dim[0] = dimLocal[0];
75  dim[1] = dimLocal[1]*up[1] + dimLocal[2]*front[1];
76  dim[2] = dimLocal[2]*front[2] + dimLocal[1]*up[2];
77  // dim contains (dx,dy,dz)
78  dx = dim[0];
79  dy = dim[1];
80  dz = dim[2];
81  osg::ref_ptr<osg::Geode> geo = new osg::Geode();
82  if ( geo.valid() ) {
83  // Box to visualize the chassis
84  osg::ref_ptr<osg::Shape> sp = new osg::Box( osg::Vec3(0.,0.,0.), dx, dy, dz);
85  if ( sp) {
86  osg::ref_ptr<osg::ShapeDrawable> sd = new osg::ShapeDrawable(sp);
87  if ( sd )
88  geo->addDrawable(sd);
89  else fprintf(stderr,"Error creating osg::Shape\n");
90  } else fprintf(stderr,"Error creating osg::Shape\n");
91  } else fprintf(stderr,"Error creating Geode\n");
92  if ( !model ) model = new osg::PositionAttitudeTransform;
93  model->addChild(geo);
94  model->setNodeMask(CastsShadowTraversalMask);
95 
96  osg::ref_ptr<osg::Material> mat = new osg::Material;
97  mat->setAmbient (osg::Material::FRONT_AND_BACK, osg::Vec4(0.4, 0.3, 0., 1.0));
98  mat->setDiffuse (osg::Material::FRONT_AND_BACK, osg::Vec4(0.8, 0.7, 0.0, 1.0));
99  mat->setSpecular(osg::Material::FRONT_AND_BACK, osg::Vec4(0.1, 0.1, 0, 1.0));
100  mat->setShininess(osg::Material::FRONT_AND_BACK, 12);
101  setMaterial(mat);
102 
103  setMass(m);
104  // Center-of-gravity is shifted by shift ????
105  btTransform shift(btQuaternion::getIdentity(), btVector3(0.f, 0.f, 0.f));
106  // Box for collisions and center-of-gravity definition
107  btCollisionShape* boxShape = new btBoxShape(osg2bt_Vec3(dim/2.));
108  btCompoundShape* chassisShape = new btCompoundShape();
109  chassisShape->addChildShape(shift, boxShape);
110  shape = chassisShape;
111  if ( !shape ) fprintf(stderr,"Error creating btShape\n");
112 
113  createRigidBody();
114 
115  rayCaster = new btDefaultVehicleRaycaster(world->dynamic);
116  btRaycastVehicle::btVehicleTuning tuning;
117  vehicle = new btRaycastVehicle(tuning, body, rayCaster);
118 
119  body->setActivationState(DISABLE_DEACTIVATION);
120  world->dynamic->addVehicle(this->vehicle);
121 
122  //Adds the wheels to the vehicle
123  btVector3 halfExtents = btVector3(dim[0]/2.,dim[1]/2.,dim[2]/2.);
124  addWheels(&halfExtents, this->vehicle, tuning);
125 
126  printf("vehicle body created\n");
127 
128  vehicle->setCoordinateSystem( 0, osg2bt_Vec3(up).maxAxis(), osg2bt_Vec3(front).maxAxis() );
129  return;

132  void addWheels(
133  btVector3* halfExtents,
134  btRaycastVehicle* vehicle,

136  {
137  // The direction of the raycast, the btRaycastVehicle uses raycasts

139  btVector3 wheelDirectionCS0(-osg2bt_Vec3(up));

141  // The axis which the wheel rotates arround
142  btVector3 wheelAxleCS( osg2bt_Vec3(front ^ up) );
143 
144  // center-of mass height if mass=0
145  // height = suspensionRestLength-mass.g/m_suspensionStiffness
146  btScalar suspensionRestLength(0.80);
147  btScalar wheelWidth(0.2);
148  btScalar wheelRadius(0.3);
149  // The height where the wheels are connected to the chassis
150  double connectionHeight = 0.4;
151  double frontAxisPos = 1.25;
152  double backAxisPos = 1.25;
153 
154  // All the wheel configuration assumes the vehicle is centered at the origin and a right handed coordinate system is used
155  btVector3 wheelConnectPoint;
156 
157  // Adds the front wheels
158  wheelConnectPoint = btVector3( halfExtents->x() + wheelWidth/2.,
159  connectionHeight*up[1] + frontAxisPos*front[1],
160  connectionHeight*up[2] + frontAxisPos*front[2]);
161  vehicle->addWheel(wheelConnectPoint, wheelDirectionCS0, wheelAxleCS, suspensionRestLength, wheelRadius, tuning, true);
162  vehicle->addWheel(wheelConnectPoint*btVector3(-1, 1, 1), wheelDirectionCS0, wheelAxleCS, suspensionRestLength, wheelRadius, tuning, true);
163 
164  //Adds the rear wheels
165  wheelConnectPoint = btVector3( halfExtents->x() + wheelWidth/2.,
166  connectionHeight*up[1] - backAxisPos*front[1],
167  connectionHeight*up[2] - backAxisPos*front[2]);
168  vehicle->addWheel(wheelConnectPoint, wheelDirectionCS0, wheelAxleCS, suspensionRestLength, wheelRadius, tuning, false);
169  vehicle->addWheel(wheelConnectPoint*btVector3(-1, 1, 1), wheelDirectionCS0, wheelAxleCS, suspensionRestLength, wheelRadius, tuning, false);
170 
171  // Create one graphics wheel

173  osg::ref_ptr<osg::Shape> sp;
174  sp = new osg::Cylinder( osg::Vec3(0.,0.,0.), wheelRadius, wheelWidth);
175  if ( sp) {
176  osg::ref_ptr<osg::ShapeDrawable> sd = new osg::ShapeDrawable(sp);
177  if ( sd ) {
178  geo->addDrawable(sd);
179  // Setup tire material

182  osg::ref_ptr<osg::Material> mat = new osg::Material;
183  mat->setAmbient (osg::Material::FRONT_AND_BACK, osg::Vec4(0., 0., 0., 1.0));
184  mat->setDiffuse (osg::Material::FRONT_AND_BACK, osg::Vec4(0.1, 0.1, 0.1, 1.0));
185  mat->setSpecular(osg::Material::FRONT_AND_BACK, osg::Vec4(0, 0, 0, 1.0));
186  mat->setShininess(osg::Material::FRONT_AND_BACK, 64);
187  stateset->setAttribute(mat, osg::StateAttribute::OVERRIDE|osg::StateAttribute::ON);

196  osg::ref_ptr<osg::ShapeDrawable> sd = new osg::ShapeDrawable(sp);

203  mat->setDiffuse (osg::Material::FRONT_AND_BACK, osg::Vec4(0.4, 0.4, 0.4, 1.0));

228  for (int i = 0; i < vehicle->getNumWheels(); i++)
229  {

231  /*

293  virtual void update()
294  {
295  // updateVehicle is Not required.
296  // Vehicle dynamics is updated in stepSimulation();
297  // updateVehicle requires frame_time that may not be available.
298  // vehicle->updateVehicle(frame_time);
299 
300  // Visual update
301  // Standard btosgObject::update() can be used.
302  btosgObject::update();
303  //logPosition();
304 
305  // Update Wheels
306  for (int i = 0; i < vehicle->getNumWheels(); i++)
307  {
308  // btRaycastVehicle::updateVehicle() calls updateWheelTransform for every wheel.
309  // Do not call vehicle->updateWheelTransform() here if btRaycastVehicle::updateVehicle() is called elsewhere.
310  //vehicle->updateWheelTransform(i,true);
311  btWheelInfo& iWheel = vehicle->getWheelInfo(i);
312  if ( wheel[i] ) {
313  osg::Vec3 iPos = bt2osg_Vec3(iWheel.m_chassisConnectionPointCS +
314  iWheel.m_raycastInfo.m_suspensionLength * iWheel.m_wheelDirectionCS);
315  wheel[i]->setPosition(iPos);
316 
317  btQuaternion steeringOrn(iWheel.m_wheelDirectionCS,-iWheel.m_steering);
318  btMatrix3x3 steeringMat(steeringOrn);
319 
320  btQuaternion rotatingOrn(iWheel.m_wheelAxleCS,-iWheel.m_rotation);
321  btMatrix3x3 rotatingMat(rotatingOrn);
322 
323  btMatrix3x3 fullMat = steeringMat*rotatingMat;
324  btQuaternion fullQuat;
325  fullMat.getRotation(fullQuat);
326  wheel[i]->setAttitude(bt2osg_Quat(fullQuat));
327  }
328  }
329  }
